namespace Darwin\Libs\Template\Extensions;

use Darwin\Helpers\URLHelper;
use Darwin\Libs\Template\PluginInterface;

class URL implements PluginInterface
{
    public function init( $template )
    {
        $template->register('uri', $this, 'uri');
    }

    public function uri($callback, $action='', $params=array(), $fullURL = false )
    {
        return URLHelper::URI($callback, $action, $params, $fullURL);
    }


}
